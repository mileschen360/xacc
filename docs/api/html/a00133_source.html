178  virtual std::shared_ptr<AcceleratorBuffer> createBuffer(
179  const std::string& varId) {
180  auto options = RuntimeOptions::instance();
181  std::string solverName = "DW_2000Q_VFYC";
182  if (options->exists("dwave-solver")) {
183  solverName = (*options)["dwave-solver"];
184  }
185  if (!availableSolvers.count(solverName)) {
186  XACCError(solverName + " is not available for creating a buffer.");
187  }
188  auto solver = availableSolvers[solverName];
189  auto buffer = std::make_shared<AQCAcceleratorBuffer>(varId, solver.nQubits);
190  storeBuffer(varId, buffer);
191  return buffer;
192  }
