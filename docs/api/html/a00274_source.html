76  bool VisitStmt(clang::Stmt *s) {
77  if (isa<IfStmt>(s)) {
78  auto ifStmt = cast<IfStmt>(s);
79  clang::LangOptions lo;
80  clang::PrintingPolicy policy(lo);
81  std::string ifStr;
82  llvm::raw_string_ostream ifS(ifStr);
83  ifStmt->printPretty(ifS, nullptr, policy);
84  // std::cout << "HELLO IF:\n" << ifS.str() << "\n";
85 
86  if (const auto binOp = llvm::dyn_cast<BinaryOperator>(
87  ifStmt->getCond())) {
88  if (binOp->getOpcode() == BO_EQ) {
89  // We have an equality check...
90  auto LHS = binOp->getLHS();
91  std::string str;
92  llvm::raw_string_ostream s(str);
93  LHS->printPretty(s, nullptr, policy);
94  // std::cout << "LHS IF: " << s.str() << "\n";
95  if (boost::contains(s.str(), cbitVarName)) {
96 
97  auto RHS = binOp->getRHS();
98  std::string rhsstr;
99  llvm::raw_string_ostream rhss(rhsstr);
100  RHS->printPretty(rhss, nullptr, policy);
101  // std::cout << "RHS IF: " << rhss.str() << "\n";
102 
103  auto thenCode = ifStmt->getThen();
104  std::string thenStr;
105  llvm::raw_string_ostream thenS(thenStr);
106  thenCode->printPretty(thenS, nullptr, policy);
107  auto then = thenS.str();
108  // std::cout << "ThenStmt:\n" << then << "\n";
109  then.erase(std::remove(then.begin(), then.end(), '\t'),
110  then.end());
111  boost::replace_all(then, "{\n", "");
112  boost::replace_all(then, "}\n", "");
113  boost::replace_all(then, " ", "");
114  boost::trim(then);
115 
116  int conditionalQubit = cbitRegToMeasuredQubit[s.str()];
117  currentConditional = std::make_shared<
118  xacc::quantum::ConditionalFunction>(
119  conditionalQubit);
120 
121  std::vector<std::string> vec;
122  boost::split(vec, then, boost::is_any_of("\n"));
123  nCallExprToSkip = vec.size();
124  // std::cout << "NCALLEXPRTOSKIP = " << nCallExprToSkip << "\n";
125  }
126  }
127  }
128  }
129  return true;
130  }
131  bool VisitDecl(clang::Decl *d) {
132  if (isa<VarDecl>(d)) {
133  auto varDecl = cast<VarDecl>(d);
134  auto varType = varDecl->getType().getAsString();
135  if (boost::contains(varType, "cbit")) {
136  cbitVarName = varDecl->getDeclName().getAsString();
137  // std::cout << "Found " << cbitVarName << "\n";
138  } else if (boost::contains(varType, "qbit")) {
139  qbitVarName = varDecl->getDeclName().getAsString();
140  boost::trim(qbitVarName);
141  // std::cout << "Found " << qbitVarName << "\n";
142  }
143  } else if (isa<FunctionDecl>(d)) {
144  auto c = cast<FunctionDecl>(d);
145 
146  // Skip first parameter since its supposed to be the qubit register
147  std::vector<xacc::InstructionParameter> parameters;
148  clang::FunctionDecl::param_iterator pBegin = c->param_begin();
149  clang::FunctionDecl::param_iterator pEnd = c->param_end();
150  for (auto i = pBegin+1; i != pEnd; ++i) {
151  auto paramName = (*i)->getNameAsString();
152  boost::trim(paramName);
153  xacc::InstructionParameter p(paramName);
154  parameters.push_back(p);
155  }
156  function = std::make_shared<xacc::quantum::GateFunction>(
157  c->getDeclName().getAsString(), parameters);
158  ir->addKernel(function);
159  }
160  return true;
161  }
162  bool VisitCallExpr(CallExpr * c) {
163  clang::LangOptions lo;
164  clang::PrintingPolicy policy(lo);
165  auto q = c->getType();
166  auto t = q.getTypePtrOrNull();
167 
168  if (t != NULL) {
169  bool isParameterizedInst = false;
170  auto fd = c->getDirectCallee();
171  auto gateName = fd->getNameInfo().getAsString();
172  auto gateRegistry = xacc::quantum::GateInstructionRegistry::instance();
173  auto availableGates = gateRegistry->getRegisteredIds();
174  std::shared_ptr<xacc::Instruction> inst;
175 
176  if (!std::any_of(availableGates.cbegin(), availableGates.cend(),
177  [=](std::string i) {return i == gateName;})) {
178 
179  if (ir->kernelExists(gateName)) {
180  inst = ir->getKernel(gateName);
181  }
182 
183  } else {
184 
185  std::vector<int> qubits;
186  std::vector<xacc::InstructionParameter> params;
187  for (auto i = c->arg_begin(); i != c->arg_end(); ++i) {
188  std::string arg;
189  llvm::raw_string_ostream argstream(arg);
190  i->printPretty(argstream, nullptr, policy);
191  auto argStr = argstream.str();
192 // std::cout << "Arg: " << argstream.str() << "\n";
193 
194  if (boost::contains(argStr, qbitVarName)) {
195  boost::replace_all(argStr, qbitVarName, "");
196  boost::replace_all(argStr, "[", "");
197  boost::replace_all(argStr, "]", "");
198  qubits.push_back(std::stoi(argStr));
199  } else {
200  // This is a gate parameter!!!
201  isParameterizedInst = true;
202 
203  // This parameter could just be a hard-coded value
204  // or it could be a reference to a variable parameter...
205  try {
206  double d = boost::lexical_cast<double>(argStr);
207  params.push_back(d);
208  } catch (const boost::bad_lexical_cast &) {
209  params.push_back(argStr);
210  }
211  }
212  }
213 
214  if (isParameterizedInst) {
215 
216  int idx = 0;
217  inst = gateRegistry->create(gateName, qubits);
218  for (auto p : params) {
219  inst->setParameter(idx, p);
220  idx++;
221  }
222 
223  } else if (gateName != "MeasZ") {
224 
225  inst = gateRegistry->create(gateName, qubits);
226  }
227 
228  }
229 
230  if (gateName != "MeasZ") {
231 
232  if (nCallExprToSkip == 0) {
233  function->addInstruction(inst);
234  } else {
235  // std::cout << "Adding Conditional Inst: " << gateName << "\n";
236  currentConditional->addInstruction(inst);
237  nCallExprToSkip--;
238 
239  if (nCallExprToSkip == 0) {
240  function->addInstruction(currentConditional);
241  }
242  }
243  }
244  }
245 
246  return true;
247  }
248  bool VisitBinaryOperator(BinaryOperator *b) {
249 
250  clang::LangOptions lo;
251  clang::PrintingPolicy policy(lo);
252 
253  if (b->isAssignmentOp()) {
254  auto rhs = b->getRHS();
255  std::string rhsstr;
256  llvm::raw_string_ostream rhss(rhsstr);
257  rhs->printPretty(rhss, nullptr, policy);
258  auto rhsString = rhss.str();
259 
260  if (boost::contains(rhsString, "MeasZ")) {
261  auto lhs = b->getLHS();
262  std::string lhsstr;
263  llvm::raw_string_ostream lhss(lhsstr);
264  lhs->printPretty(lhss, nullptr, policy);
265  auto lhsString = lhss.str();
266 
267  boost::replace_all(lhsString, cbitVarName, "");
268  boost::replace_all(lhsString, "[", "");
269  boost::replace_all(lhsString, "]", "");
270 
271  boost::replace_all(rhsString, "MeasZ", "");
272  boost::replace_all(rhsString, "(", "");
273  boost::replace_all(rhsString, ")", "");
274  boost::replace_all(rhsString, qbitVarName, "");
275  boost::replace_all(rhsString, "[", "");
276  boost::replace_all(rhsString, "]", "");
277 
278  // lhsString now just contains the classical index bit
279  auto inst =
280  xacc::quantum::GateInstructionRegistry::instance()->create(
281  "Measure",
282  std::vector<int> { std::stoi(rhsString) });
283  xacc::InstructionParameter classicalIdx(std::stoi(lhsString));
284  inst->setParameter(0, classicalIdx);
285 
286  cbitRegToMeasuredQubit.insert(
287  std::make_pair(lhss.str(), std::stoi(rhsString)));
288 
289  function->addInstruction(inst);
290 
291  }
292  }
293 
294  return true;
