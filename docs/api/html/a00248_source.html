117  void visit(ConditionalFunction& c) {
118  auto visitor = std::make_shared<QuilVisitor>();
119  auto classicalBitIdx = qubitToClassicalBitIndex[c.getConditionalQubit()];
120  quilStr += "JUMP-UNLESS @" + c.getName() + " [" + std::to_string(classicalBitIdx) + "]\n";
121  for (auto inst : c.getInstructions()) {
122  inst->accept(visitor);
123  }
124  quilStr += visitor->getQuilString();
125  quilStr += "LABEL @" + c.getName() + "\n";
126  }
