40 template<typename InputIterator, typename OutputIterator, typename Predicate>
41 OutputIterator copy_if(InputIterator first, InputIterator last,

59 inline std::vector<std::string> tokenize(const std::string& input) {
60  typedef boost::escaped_list_separator<char> separator_type;
61  separator_type separator("\\", "= ", "\"\'");
62  // Tokenize the input.
63  boost::tokenizer<separator_type> tokens(input, separator);
64  // Copy non-empty tokens from the tokenizer into the result.
65  std::vector<std::string> result;
66  xacc::copy_if(tokens.begin(), tokens.end(), std::back_inserter(result),
67  !boost::bind(&std::string::empty, _1));
68  return result;
69 }
