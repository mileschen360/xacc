186  virtual void evaluateVariableParameters(
187  std::vector<InstructionParameter> runtimeParameters) {

189  std::map<std::string, InstructionParameter> varToValMap;
190 
191  int i = 0;
192  for (auto funcParam : parameters) {
193  varToValMap.insert(
194  std::make_pair(boost::get<std::string>(funcParam),
195  runtimeParameters[i]));
196  i++;
197  }
198 
199  for (const auto& gateIdVarName : cachedVariableInstructions) {
200  auto inst = getInstruction(gateIdVarName.first);
201  auto varInstDependsOn = gateIdVarName.second.second;
202  auto instParamIdx = gateIdVarName.second.first;
203 
204  int indexOfRuntimeParam, counter = 0;
205  for (auto p : parameters) {
206  if (boost::get<std::string>(p) == varInstDependsOn) {
207  indexOfRuntimeParam = counter;
208  break;
209  }
210  counter++;
211  }
212 
213  inst->setParameter(instParamIdx, runtimeParameters[indexOfRuntimeParam]);
214  }
215 
216  i = 0;
217  for (auto inst : instructions) {
218  if (inst->isComposite()) {
219  std::dynamic_pointer_cast<Function>(inst)->evaluateVariableParameters(
220  runtimeParameters);
221  } else if (inst->isParameterized() && inst->getName() != "Measure") {
222 
223  for (int j = 0; j < inst->nParameters(); ++j) {
224  auto instParam = inst->getParameter(j);
225 
226  if (instParam.which() == 3) {
227  // This is a variable
228  auto variable = boost::get<std::string>(instParam);
229 
230  auto runtimeParameter = varToValMap[variable];
231 
232  inst->setParameter(j, runtimeParameter);
233 
234  cachedVariableInstructions.insert(std::make_pair(i, std::make_pair(j, variable)));
235  }
236  }
237  }
238  i++;
239  }
240  }
