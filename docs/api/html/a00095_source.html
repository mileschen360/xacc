97  void parse(int argc, char** argv) {
98 
99  // Get a reference to the RuntimeOptions
100  auto runtimeOptions = RuntimeOptions::instance();
101 
102  // Load all default Compilers and Accelerators,
103  // ie those in XACC INSTALL DIR/lib
104  boost::filesystem::path xaccPath(
105  std::string(XACC_INSTALL_DIR) + std::string("/lib"));
106  if (boost::filesystem::exists(xaccPath)) {
107  boost::filesystem::directory_iterator end_itr;
108 
109  // cycle through the directory
110  for (boost::filesystem::directory_iterator itr(xaccPath);
111  itr != end_itr; ++itr) {
112  auto p = itr->path();
113  if (p.extension() == ".so"
114  && !boost::contains(p.string(), "xacc-quantum-aqc")
115  && !boost::contains(p.string(), "xacc-quantum-gate")) {
116  namespace dll = boost::dll;
117  dll::shared_library lib(p,
118  dll::load_mode::append_decorations);
119  if (lib.has("registerCompiler")) {
120  typedef void (RegisterCompiler)();
121  auto regFunc =
122  boost::dll::import_alias<RegisterCompiler>(p,
123  "registerCompiler",
124  dll::load_mode::append_decorations);
125  regFunc();
126  }
127  if (lib.has("registerAccelerator")) {
128  typedef void (RegisterAccelerator)();
129  auto regFunc = boost::dll::import_alias<
130  RegisterAccelerator>(p, "registerAccelerator",
131  dll::load_mode::append_decorations);
132  regFunc();
133  }
134  }
135  }
136  }
137 
138  auto registeredAccOptions = AcceleratorRegistry::instance()->getRegisteredOptions();
139  auto registeredCompilerOptions = CompilerRegistry::instance()->getRegisteredOptions();
140 
141  for (auto s : registeredAccOptions) {
142  xaccOptions->add(*s.get());
143  }
144  for (auto s : registeredCompilerOptions) {
145  xaccOptions->add(*s.get());
146  }
147 
148  // Parse the command line options
149  variables_map clArgs;
150  store(parse_command_line(argc, argv, *xaccOptions.get()), clArgs);
151  if (clArgs.count("help")) {
152  std::cout << *xaccOptions.get() << "\n";
153  XACCInfo(
154  "\n[xacc] XACC Finalizing\n[xacc::compiler] Cleaning up Compiler Registry."
155  "\n[xacc::accelerator] Cleaning up Accelerator Registry.");
156  xacc::CompilerRegistry::instance()->destroy();
157  xacc::AcceleratorRegistry::instance()->destroy();
158  exit(0);
159  }
160 
161  // If the user provides a path to a compiler plugin,
162  // then load it
163  if (clArgs.count("load-compiler")) {
164  auto loadPath = clArgs["load-compiler"].as<std::string>();
165  boost::filesystem::path p(loadPath);
166  namespace dll = boost::dll;
167  dll::shared_library lib(p, dll::load_mode::append_decorations);
168  if (lib.has("registerCompiler")) {
169  typedef void (RegisterCompiler)();
170  auto regFunc = boost::dll::import_alias<RegisterCompiler>(p,
171  "registerCompiler", dll::load_mode::append_decorations);
172  regFunc();
173  }
174  clArgs.erase("load-compiler");
175  }
176 
177  // If the user provides a path to a accelerator plugin,
178  // then load it
179  if (clArgs.count("load-accelerator")) {
180  auto loadPath = clArgs["load-accelerator"].as<std::string>();
181  boost::filesystem::path p(loadPath);
182  namespace dll = boost::dll;
183  dll::shared_library lib(p, dll::load_mode::append_decorations);
184  if (lib.has("registerAccelerator")) {
185  typedef void (RegisterAccelerator)();
186  auto regFunc = boost::dll::import_alias<RegisterAccelerator>(p,
187  "registerAccelerator", dll::load_mode::append_decorations);
188  regFunc();
189  }
190  clArgs.erase("load-accelerator");
191  }
192 
193  bool listTypes = false;
194  if (clArgs.count("list-compilers")) {
195  auto ids = CompilerRegistry::instance()->getRegisteredIds();
196  XACCInfo("Available XACC Compilers:");
197  for (auto i : ids) {
198  XACCInfo("\t" + i);
199  }
200  listTypes = true;
201  }
202 
203  if (clArgs.count("list-accelerators")) {
204  auto ids = AcceleratorRegistry::instance()->getRegisteredIds();
205  XACCInfo("Available XACC Accelerators:");
206  for (auto i : ids) {
207  XACCInfo("\t" + i);
208  }
209  listTypes = true;
210  }
211 
212  if (listTypes) {
213  XACCInfo(
214  "\n[xacc] XACC Finalizing\n[xacc::compiler] Cleaning up Compiler Registry."
215  "\n[xacc::accelerator] Cleaning up Accelerator Registry.");
216  xacc::CompilerRegistry::instance()->destroy();
217  xacc::AcceleratorRegistry::instance()->destroy();
218  exit(0);
219  }
220 
221  // Add all other string options to the global runtime option
222  for (auto it = clArgs.begin(); it != clArgs.end();
223  ++it) {
224  runtimeOptions->insert(
225  std::make_pair(it->first, it->second.as<std::string>()));
226  }
227  }
228 
229  void addStringOption(const std::string key, const std::string description = "") {
230  xaccOptions->add_options()(key.c_str(), value<std::string>(), description.c_str());
231  }
