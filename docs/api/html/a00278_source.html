137  void visit(ConditionalFunction& c) {

139  auto classicalBitIdx = qubitToClassicalBitIndex[c.getConditionalQubit()];

163  void visit(CPhase& cp) {

165  auto targetQubit = cp.bits()[1];
166  auto params = cp.getParameters();
167 
168  // Check if this is a string

171  } else {

174 
175  scaffoldStr += "// BEGIN CPHASE GATE\n";
176  baseParameterizedInstruction("Rz", std::vector<int>{targetQubit}, params);
177  baseInstruction("CNOT", std::vector<int>{targetQubit, srcQubit});
