142  ChimeraGraph(int gridSize) : DWGraph(8 * gridSize * gridSize) {
143  // We are going to assign a tuple of ints to each vertex
144  std::map<std::tuple<int, int, int, int>, int> vertexTuples;
145  int i, j, k, l, v = 0;
146 
147  // Loop through the list of vertices and give
148  // them the correct 4-tuple: (i,j) is unit cell in
149  // the chimera grid, (k,l) is vertex in unit cell, k = 0-3,
150  // l = 0,1 (left, right of bipartite graph)
151  for (i = 0; i < gridSize; i++) {
152  for (j = 0; j < gridSize; j++) {
153  for (l = 0; l < 2; l++) {
154  for (k = 0; k < 4; k++) {
155  vertexTuples.insert(
156  std::make_pair(std::make_tuple(i, j, k, l), v));
157  v++;
158  }
159  }
160  }
161  }
162 
163  // Setup edges within each cell
164  for (i = 0; i < gridSize; i++) {
165  for (j = 0; j < gridSize; j++) {
166  for (k = 0; k < 4; k++) {
167  for (int k2 = 0; k2 < 4; k2++) {
168  int v1 = vertexTuples[std::make_tuple(i, j, k, 0)];
169  int v2 = vertexTuples[std::make_tuple(i, j, k2, 1)];
170  addEdge(v1, v2);
171  }
172 
173  }
174  }
175  }
176 
177  // Setup edges between cells
178  for (i = 0; i < gridSize - 1; i++) {
179  for (j = 0; j < gridSize; j++) {
180  for (k = 0; k < 4; k++) {
181  int v1 = vertexTuples[std::make_tuple(i, j, k, 0)];
182  int v2 = vertexTuples[std::make_tuple(i + 1, j, k, 0)];
183  addEdge(v1, v2);
184 
185  }
186  }
187  }
188 
189  for (i = 0; i < gridSize; i++) {
190  for (j = 0; j < gridSize - 1; j++) {
191  for (k = 0; k < 4; k++) {
192  int v1 = vertexTuples[std::make_tuple(i, j, k, 1)];
193  int v2 = vertexTuples[std::make_tuple(i, j + 1, k, 1)];
194  addEdge(v1, v2);
195 
196  }
197  }
198  }
199  }
200 
201  virtual ~ChimeraGraph() {}
202 };
