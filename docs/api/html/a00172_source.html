76  std::string write() {
77  writer->StartObject();
78 
79  writer->String("kernels");
80  writer->StartArray();
81  for (auto f : functions) {
82  // This is a Function, start it as an Object
83  writer->StartObject();
84 
85  writer->String("function");
86  writer->String(f->getName());
87 
88  // All functions have instructions, start
89  // that array here.
90  writer->String("instructions");
91  writer->StartArray();
92 
93  topLevelInstructionIterator = std::make_shared<
94  xacc::InstructionIterator>(f);
95  while (topLevelInstructionIterator->hasNext()) {
96  // Get the next node in the tree
97  auto nextInst = topLevelInstructionIterator->next();
98  nextInst->accept(this);
99  }
100 
101  // End Instructions
102  writer->EndArray();
103 
104  // End Function
105  writer->EndObject();
106  }
107  writer->EndArray();
108  writer->EndObject();
109  return buffer->GetString();
110  }

215  auto cnAsPtr = std::make_shared<ConditionalFunction>(cn);

217  xacc::InstructionIterator it(cnAsPtr);
218  it.next();
219  while (it.hasNext()) {
220  // Get the next node in the tree

225  // End Instructions
226  writer->EndArray();
227  writer->EndObject();
