56  static QuantumCircuit getCircuitGraph(
57  const std::string& flatQasmStr) {
58 
59  // Local Declarations
60  QuantumCircuit graph;
61  std::map<std::string, int> qubitVarNameToId;
62  std::vector<std::string> qasmLines;
63  std::vector<int> allQbitIds;
64  std::regex newLineDelim("\n"), spaceDelim(" ");
65  std::regex qubitDeclarations("\\s*qubit\\s*\\w+");
66  std::sregex_token_iterator first{flatQasmStr.begin(), flatQasmStr.end(), newLineDelim, -1}, last;
67  int nQubits = 0, qbitId = 0, layer = 1, gateId = 1;
68  std::string qubitVarName;
69  qasmLines = {first, last};
70 
71  // Let's now loop over the qubit declarations,
72  // and construct a mapping of qubit var names to integer id,
73  // and get the total number of qubits
74  for (auto i = std::sregex_iterator(flatQasmStr.begin(), flatQasmStr.end(),
75  qubitDeclarations); i != std::sregex_iterator(); ++i) {
76  std::string qubitLine = (*i).str();
77  qubitLine.erase(std::remove(qubitLine.begin(), qubitLine.end(), '\n'), qubitLine.end());
78  std::sregex_token_iterator first{qubitLine.begin(), qubitLine.end(), spaceDelim, -1}, last;
79  std::vector<std::string> splitQubitLine = {first, last};
80  qubitVarNameToId[splitQubitLine[1]] = qbitId;
81  splitQubitLine[1].erase(
82  std::remove_if(splitQubitLine[1].begin(), splitQubitLine[1].end(), &isdigit),
83  splitQubitLine[1].end());
84  qubitVarName = splitQubitLine[1];
85  allQbitIds.push_back(qbitId);
86  qbitId++;
87  }
88 
89  // Set the number of qubits
90  nQubits = qubitVarNameToId.size();
91 
92  // First create a starting node for the initial
93  // wave function - it should have nQubits outgoing
94  // edges
95  graph.addVertex("InitialState", 0, 0, allQbitIds, true, std::vector<std::string>{});
96 
97  std::vector<CircuitNode> gateOperations;
98  for (auto line : qasmLines) {
99  // If this is a gate line...
100  if (!boost::contains(line, "qubit") && !boost::contains(line, "cbit")) {
101 
102  // Create a new CircuitNode
103  CircuitNode node;
104 
105  // Split the current qasm command at the spaces
106  std::sregex_token_iterator first { line.begin(),
107  line.end(), spaceDelim, -1 }, last;
108  std::vector<std::string> gateCommand = {first, last};
109 
110  // Set the gate as a lowercase gate name string
111  auto g = boost::to_lower_copy(gateCommand[0]);
112  boost::trim(g);
113  if (g == "measz") g = "measure";
114  std::get<0>(node.properties) = g;
115 
116  // If not a 2 qubit gate, and if the acting
117  // qubit is different than the last one, then
118  // keep the layer the same, otherwise increment
119  if (incrementLayer(gateCommand, qubitVarNameToId,
120  gateOperations, layer)) {
121  layer++;
122  }
123 
124  // Set the current layer
125  std::get<1>(node.properties) = layer;
126 
127  // Set the gate vertex id
128  std::get<2>(node.properties) = gateId;
129  gateId++;
130 
131  // Set the qubits this gate acts on
132  std::vector<int> actingQubits;
133  if (!boost::contains(gateCommand[1], ",")) {
134  actingQubits.push_back(qubitVarNameToId[gateCommand[1]]);
135  } else {
136 
137 
138  // FIXME Need to differentiate between qubits and parameters here
139  // First we need the qubit register variable name
140 
141  int counter = 0;
142  std::vector<std::string> splitComma, props;
143  boost::split(splitComma, gateCommand[1], boost::is_any_of(","));
144  for (auto segment : splitComma) {
145  if (boost::contains(segment, qubitVarName)) {
146  actingQubits.push_back(qubitVarNameToId[segment]);
147  } else {
148  // This is not a qubit, it must be a parameter for gate
149  props.push_back("PARAM_" + std::to_string(counter));
150  std::get<5>(node.properties) = props;
151  counter++;
152  }
153  }
154  }
155 
156  // Set the acting qubits
157  std::get<3>(node.properties) = actingQubits;
158 
159  // Add this gate to the local vector
160  // and to the graph
161  gateOperations.push_back(node);
162  graph.addVertex(node);
163  }
164  }
165 
166  // Add a final layer for the graph sink
167  CircuitNode finalNode;
168  std::get<0>(finalNode.properties) = "FinalState";
169  std::get<1>(finalNode.properties) = layer+1;
170  std::get<2>(finalNode.properties) = gateId;
171  std::get<3>(finalNode.properties) = allQbitIds;
172  std::get<4>(finalNode.properties) = true;
173 
174  graph.addVertex(finalNode);
175  gateOperations.push_back(finalNode);
176 
177  // Set how many layers are in this circuit
178  int maxLayer = layer+1;
179 
180  generateEdgesFromLayer(1, graph, gateOperations, 0);
181 
182  return graph;
183  }

194  static void linkConditionalQasm(QuantumCircuit& mainGraph,
195  std::vector<QuantumCircuit>& conditionalGraphs,
196  std::vector<int>& conditionalQubits) {
197 
198  // At this point we have a main circuit graph,
199  // and one or more smaller conditional graphs (each with
200  // initial and final state nodes.
201 
202  // We need to loop through the conditional graphs and
203  // pull out the gate vertices (skip initial final state nodes)
204  // and add them to the main graph after some conditional nodes
205 
206  assert (conditionalQubits.size() == conditionalGraphs.size());
207 
208  int counter = 0;
209  int finalIdMainGraph = mainGraph.getVertexProperty<2>(mainGraph.order() - 1);
210  int finalLayerMainGraph = mainGraph.getVertexProperty<1>(mainGraph.order() - 1);
211  int layer = finalLayerMainGraph + 1;
212  int id = finalIdMainGraph + 1;
213 
214  // By default all conditional graph nodes should be disabled
215  // Conditional nodes should have acting qubits set to the
216  // measured qubit, and gate vertex id set to the measurement
217  // gate vertex id.
218  for (auto g : conditionalGraphs) {
219  CircuitNode node;
220  std::get<0>(node.properties) = "conditional";
221  std::get<2>(node.properties) = id;
222  std::get<3>(node.properties) = std::vector<int> {conditionalQubits[counter]};
223  mainGraph.addVertex(node);
224 
225  // Connect the main graph to the cond node
226  mainGraph.addEdge(finalIdMainGraph, id);
227 
228  id++;
229 
230  std::vector<CircuitNode> newGates;
231  int initialStateId, initialStateLayer;
232  for (int i = 0; i < g.order(); i++) {
233  CircuitNode newVert;
234  newVert.properties = g.getVertexProperties(i);
235  std::get<0>(newVert.properties) = g.getVertexProperty<0>(i);
236  std::get<1>(newVert.properties) = layer+g.getVertexProperty<1>(i);
237  std::get<2>(newVert.properties) = id;
238  std::get<3>(newVert.properties) = g.getVertexProperty<3>(i);
239  std::get<4>(newVert.properties) = false;
240  mainGraph.addVertex(newVert);
241  newGates.push_back(newVert);
242 
243  // Connect conditional node with first node here
244  if(i == 0) mainGraph.addEdge(id - 1, id);
245  if(i == 0) initialStateId = id;
246  if (i == 0) initialStateLayer = std::get<1>(newVert.properties);
247 
248  id++;
249  }
250 
251  generateEdgesFromLayer(initialStateLayer+1, mainGraph, newGates, initialStateId);
252 
253  // Set the new layer for the next conditional node
254  layer = mainGraph.getVertexProperty<1>(id - 1) + 1;
255  counter++;
256  }

274  static void generateEdgesFromLayer(const int layer, QuantumCircuit& graph,
275  std::vector<CircuitNode>& gateOperations, int initialStateId) {
276 
277  int nQubits = std::get<3>(graph.getVertexProperties(0)).size();
278  int maxLayer = std::get<1>(
279  gateOperations[gateOperations.size() - 1].properties);
280 
281  std::map<int,int> qubitToCurrentGateId;
282  for (int i = 0 ; i < nQubits; i++) {
283  qubitToCurrentGateId[i] = initialStateId;
284  }
285 
286  int currentLayer = layer;
287  while (currentLayer <= maxLayer) {
288  std::vector<CircuitNode> currentLayerGates;
289  std::copy_if(gateOperations.begin(), gateOperations.end(),
290  std::back_inserter(currentLayerGates),
291  [&](const CircuitNode& c) {return std::get<1>(c.properties) == currentLayer;});
292 
293  for (auto n : currentLayerGates) {
294  std::vector<int> actingQubits = std::get<3>(n.properties);
295  for (auto qubit : actingQubits) {
296  int currentQubitGateId = qubitToCurrentGateId[qubit];
297  graph.addEdge(currentQubitGateId, std::get<2>(n.properties));
298  qubitToCurrentGateId[qubit] = std::get<2>(n.properties);
299  }
300  }
301 
302  currentLayer++;
303  }
304  }
314  static bool incrementLayer(const std::vector<std::string>& gateCommand,
315  std::map<std::string, int>& qubitVarNameToId,
316  const std::vector<CircuitNode>& gates, const int& currentLayer) {
317 
318  bool oneQubitGate = !boost::contains(gateCommand[1], ","), noGateAtQOnL = true;
319  auto g = boost::to_lower_copy(gateCommand[0]);
320  boost::trim(g);
321  if (g == "measz") g = "measure";
322 
323 
324  std::vector<CircuitNode> thisLayerGates;
325  std::copy_if(gates.begin(), gates.end(),
326  std::back_inserter(thisLayerGates),
327  [&](const CircuitNode& c) {return std::get<1>(c.properties) == currentLayer;});
328 
329  for (auto layerGate : thisLayerGates) {
330  std::vector<int> qubits = std::get<3>(layerGate.properties);
331  for (auto q : qubits) {
332  if (qubitVarNameToId[gateCommand[1]] == q) {
333  noGateAtQOnL = false;
334  }
335  }
336  }
337 
338  if (!oneQubitGate) {
339  return true;
340  } else if (!noGateAtQOnL) {
341  return true;
342  } else if (!gates.empty()
343  && (std::get<0>(gates[gates.size() - 1].properties)
344  == "measure") && g != "measure") {
345  return true;
346  }
347 
348  return false;
349  }
350 };
