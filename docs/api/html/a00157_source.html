160  void operator()(std::ostream& out, const BoostVertex& v) const {
161  auto node = vertex(v, graph);
162  std::stringstream ss;
163  ss << graph[node].properties;
164 
165  // Now we have a string of comma separated parameters
166  std::string result;
167  auto vertexString = ss.str();
168  boost::trim(vertexString);
169  std::vector<std::string> splitVec;

172  int counter = 0;
173  for (std::size_t i = 0; i < splitVec.size(); i++) {
174  if (!graph[node].propertyNames[counter].empty()) {
175  auto s = splitVec[i];
176  if (i == 0) {
177  s.insert(8, graph[node].propertyNames[counter] + "=");
178  } else {
179  s.insert(0, graph[node].propertyNames[counter] + "=");
180  }
181  counter++;
182  result += s + ",";
183  }
184  }
185 
186  result = result.substr(0,result.size() - 1);
187  if (result.substr(result.size() - 2, result.size()) != "\"]") {
188  result += "\"]";
189  }
190  out << " " << result;
191  }

393  int diameter() {
394  // Set some aliases we'll need
395  using DistanceProperty = boost::exterior_vertex_property<adj_list, int>;
396  using DistanceMatrix = typename DistanceProperty::matrix_type;
397  using DistanceMatrixMap = typename DistanceProperty::matrix_map_type;
398  using EccentricityProperty = boost::exterior_vertex_property<adj_list, int>;
399  using EccentricityContainer = typename EccentricityProperty::container_type;
400  using EccentricityMap = typename EccentricityProperty::map_type;
401 
402  // Construct the distance mapping
403  DistanceMatrix distances(order());
404  DistanceMatrixMap dm(distances, *_graph.get());
405  constant_property_map<edge_type, int> wm(1);
406 
407  // Compute the shortest paths
408  floyd_warshall_all_pairs_shortest_paths(*_graph.get(), dm, weight_map(wm));
409 
410  // Now get diameter information
411  EccentricityContainer eccs(order());
412  EccentricityMap em(eccs, *_graph.get());
413 
414  // Return the diameter
415  return all_eccentricities(*_graph.get(), dm, em).second;
416  }
